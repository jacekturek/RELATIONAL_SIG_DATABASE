# Do some relational math - queries

Relations = tables have attributes:
SELECT DISTINCT will show RANKINGS (IN ME) in the database:

SELECT DISTINCT [RANKING IN ME]
  FROM [SEO_SIG_IN_SOC_MEDIA].[dbo].[RANKINGS]
